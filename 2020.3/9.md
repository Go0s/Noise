- nginx反向代理泛解析的域名：

  ```
  server {
    resolver 8.8.8.8;
    listen 80;
    
    server_name *.xxx.com;
    if ($host ~* "(.*)\.xxx.com") {
    	set $subdomain $1; 
    }
    location / {
    	proxy_set_header Referer https://$subdomain.yyy.com;
    	proxy_ pass "https://$subdomain.yyy.com";
    }
  }
  ```
